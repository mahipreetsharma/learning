Parameter passing
    void swap(int *x,int *y){ //call by header
        int temp;
        temp = *x;
        *x = *y;
        *y = temp;
    }

    int main(){
        int a, b;
        int a =10;
        int b =12;
        swap(&a,&b);
    }

    void swap(int &x, int &y){//call by reference
        int temp;
        temp = x;
        x = y;
        y = temp;
    }
    int main(){
        int a,b;
        a =10;
        b =20;
        swap(a,b);
    }
